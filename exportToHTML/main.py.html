import copy

player, AI = 'x', 'o'

def minimax(board, depth, max_player):
    score = get_score(board, max_player)
    if score == 10:
        return 10
    if score == -10:
        return -10

    if (max_player):
        best_score = -1000

        for i in range(len(board)):
            for j in range(len(board)):
                if board[i][j] == '-':
                    board[i][j] = player
                    block(board, i, j)
                    best_score= max(best_score, minimax(board, depth + 1, not max_player))
                    unblock(board, i, j)
        return best_score
    else:
        best_score = 1000
        for i in range(len(board)):
            for j in range(len(board)):
                if board[i][j] == '-':
                    board[i][j] = AI
                    block(board, i, j)
                    best_score= min(best_score, minimax(board, depth + 1, not max_player))
                    unblock(board, i, j)
        return best_score

def check_terminal(state):
    for i in range(len(state)):
        for j in range(len(state)):
            if state[i][j] == '-':
                return False
    return True

def get_score(board, max_player):
    if check_terminal(board) == True:
        if max_player == False:
            return 10
        else: return -10
    else:
        return 0

def block(state, r,  c):
    for i in range(len(state)):
        if i == r - 1 or i == r or i == r + 1:
            for j in range(len(state)):
                if (j == c - 1 or j == c or j == c + 1) and state[i][j] == '-':
                    state[i][j] = '/'
    return state

def unblock(state, r,  c):
    for i in range(len(state)):
        if i == r - 1 or i == r or i == r + 1:
            for j in range(len(state)):
                if j == c - 1 or j == c or j == c + 1:
                    state[i][j] = '-'
    return state
